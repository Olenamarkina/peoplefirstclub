    // <!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Animated Map</title>

    <link rel="stylesheet" href="https://unpkg.com/leaflet/dist/leaflet.css" />

    <style>

        #map {

            height: 100vh;

        }

    </style>

</head>

<div id="map"></div>


<script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

<script>


    var locations = [

        { name: 'Kyiv, Ukraine', coordinates: [50.4501, 30.5234] },

        { name: 'Budapest, Hungary', coordinates: [47.4979, 19.0402] },

        { name: 'Bangkok, Thailand', coordinates: [13.7563, 100.5018] },

        { name: 'Mombasa, Kenya', coordinates: [-4.0435, 39.6682] },

        { name: 'Rokiskis, Lithuania', coordinates: [55.9574, 25.5948] },

        { name: 'Kyiv, Ukraine', coordinates: [50.4501, 30.5234] },

    ];


    var map = L.map('map').setView([0, 0], 2);


    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

        attribution: '&copy; OpenStreetMap contributors'

    }).addTo(map);


    var marker = L.marker(locations[0].coordinates).addTo(map);

    var currentIndex = 0;


    function animateMap() {

        if (currentIndex < locations.length - 1) {

            currentIndex++;

            marker.setLatLng(locations[currentIndex].coordinates);

            map.panTo(locations[currentIndex].coordinates, { animate: true, duration: 2 });

            setTimeout(animateMap, 3000); // Adjust the delay as needed

        }

    }


    animateMap();


</script>
